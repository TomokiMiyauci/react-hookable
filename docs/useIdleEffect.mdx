# useIdleEffect

Effect for track whether the user is being inactive

## Usage

```tsx
import { useIdleEffect } from 'react-hookable'
useIdleEffect({
  onIdle: () => {
    // call on Idle
  },
  onActive: () => {
    // call on active from idle
  },
  onInteract: () => {
    // call on user interaction
  }
})
```
