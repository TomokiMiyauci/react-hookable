# useFetchEffect

Effect for `fetch` API that abort automatically when unmount

## Usage

```tsx
import { useFetchEffect } from 'react-hookable'
useFetchEffect(
  {
    input: 'URL',
    onResolve: (res) => {},
    onAbort: (e) => {},
    onReject: (e) => {},
    onFinally: () => {}
  },
  deps,
  condition
)
```
