# useFetch

Safe fetch hooks that abort http request on unmount

## Usage

```tsx
import { useFetch } from 'react-hookable'
const { use: fetch } = useFetch()

fetch(input, requestInit).catch((e: Error) => {
  if (e.name === 'AbortError') {
    // catch abort error
  }
})

// unmount
// fetch will abort automatically
```
